    "ВЫБРАТЬ
    |   Ссылка
    |ИЗ
    |   Справочник.Пользователи
    |ГДЕ

    Запрос.УстановитьПараметр("Логин", ЭтотОбъект.Наименование);
    Запрос.УстановитьПараметр("Текущая", ЭтотОбъект.Ссылка);

    Если Запрос.Выполнить().Выбрать().Количество() > 0 Тогда
        Отказ = Истина;
