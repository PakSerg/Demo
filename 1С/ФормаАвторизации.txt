
&НаСервере
Функция ВойтиНаСервере()
	ПользовательНайден = Справочники.Пользователи.НайтиПоНаименованию(Логин, Истина);
	
	Если ПользовательНайден.Пустая() Тогда
		Возврат Неопределено;
	КонецЕсли;
	
	Если ПользовательНайден.Пароль <> Пароль Тогда
		Возврат Неопределено;
	КонецЕсли;
	
	Если ПользовательНайден.ДатаВхода <> Дата("01.01.0001 0:00:00") Тогда
		Если ТекущаяДата() > ДобавитьМесяц(НачалоМесяца(ПользовательНайден.ДатаВхода), 1) Тогда
			ИзменитьДанныеНаСервере(ПользовательНайден, "Активность", Ложь);
			Сообщить("Вы заблокированы");
		КонецЕсли;
	КонецЕсли;
	
	Возврат ПользовательНайден.Роль;
КонецФункции

&НаКлиенте
Процедура Войти(Команда)
	Роль = ВойтиНаСервере();
	
	Если Роль = Неопределено Тогда
		Если ПопыткиПользователя(ПользовательНайден) > 3 Тогда
			ИзменитьДанныеНаСервере(ПользовательНайден, "Активность", Ложь);
			Сообщить("Вы заблокированы");
		Иначе   
			Сообщить("Вы ввели неверный логин или пароль");
		КонецЕсли;
		
		Возврат;
	КонецЕсли;      
	
	Если НЕ АктивностьПользователя(ПОльзовательНайден) Тогда
		Сообщить("Вы заблокированы");
		Возврат;
	КонецЕсли;
	
	ИзменитьДанныеНаСервере(ПользовательНайден, "КоличествоПопыток", 0);
	
	Если ВходПользователя(ПользовательНайден) = Дата("01.01.0001 0:00:00") Тогда
		Открытьформу("ОбщаяФорма.ФормаСменыПароля", 
			Новый Структура("ПараметрПользователь", ПользовательНайден)
		);
		Сообщить("Измените пароль");
		Этаформа.Закрыть();
		Возврат; 
	КонецЕсли;
	
	ИзменитьДанныеНаСервере(ПользовательНайден, "ДатаВхода", ТекущаяДата());
	
	Если Роль = ПредопределенноеЗначение("Перечисление.Роли.Пользователь") Тогда
		Открытьформу("ОбщаяФорма.ФормаПользователя");
		Этаформа.Закрыть();
	Иначе 
		Открытьформу("Справочник.Пользователи.ФормаСписка");
	КонецЕсли;
	
	Сообщить("Вы успешно вошли");
	Этаформа.Закрыть();
	
КонецПроцедуры
